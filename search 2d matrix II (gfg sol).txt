#include <bits/stdc++.h>
using namespace std;

int solve(vector<vector<int>>&res, int n, int m, int target){
    int i=0, j=m-1;			//START FROM TOP RIGHT CORNER
    while(i<n && j>=0){
        if(res[i][j]==target)
            return 1;
        if(res[i][j]>target)
            j--;
        else i++;
    }
    return 0;
}

int main() {
	int t;
	cin>>t;
	while(t--){
	    int target,n,m;
	    cin>>n;
	    cin>>m;
	    vector<vector<int>>res(n, vector<int>(m));
	    for(int i=0; i<n;i++){
	        for(int j=0; j<m; j++){
	            cin>>res[i][j];
	        }
	    }
	    cin>>target;
	    cout<<solve(res,n,m, target)<<endl;
	}
	return 0;
}
